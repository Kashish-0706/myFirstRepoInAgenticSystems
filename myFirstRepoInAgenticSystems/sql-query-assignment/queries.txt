id INT,
name TEXT,
score INT,
passed BOOLEAN,
class INT
);

--1. Retrieve all columns
SELECT * FROM students;
--2. Retrieve only name and score
SELECT name, score FROM students;
--3. Students who scored above 80
SELECT * FROM students WHERE score > 80;
--4. Students scored above 80 AND passed
SELECT * FROM students WHERE score > 80 AND passed = TRUE;
--5. Score above 85 OR class 9
SELECT * FROM students WHERE score > 85 OR class = 9;
--6. Students who are NOT passed
SELECT * FROM students WHERE NOT passed = TRUE;
--7. Find DISTINCT class
SELECT DISTINCT class FROM students;
--8. Top 5 highest scoring students
SELECT * FROM students ORDER BY score DESC LIMIT 5;
--9. Sorted by class (ASC) and score (DESC)
SELECT * FROM students ORDER BY class ASC, score DESC;
--10. Using an alias to convert score to final_score
SELECT score AS final_score FROM students;
--11. Top 3 students who scored above 75
SELECT name, score FROM students WHERE score > 75 ORDER BY score DESC LIMIT 3;
