CREATE TABLE users (
user_id INT PRIMARY KEY,
name TEXT,
signup_date DATE
);

CREATE TABLE orders (
order_id INT PRIMARY KEY,
user_id INT,
amount INT
);

CREATE TABLE activity(
user_id INT,
login_count INT
);

--1. Retrieve each user's name and amount (INNER JOIN)
SELECT u.name, o.amount
FROM users u INNER JOIN orders o 
ON u.user_id = o.user_id;

--2. Retrieve user name and total order amount 
SELECT u.name, SUM(o.amount) AS total_order_amount
FROM users u
INNER JOIN orders o ON u.user_id = o.user_id
GROUP BY u.name;

--3. Retrieve all users and their login counts
SELECT u.name, a.login_count FROM users u
LEFT JOIN activity a ON u.user_id = a.user_id;

--4. Retrieve users who never placed an order
SELECT u.name FROM users u
LEFT JOIN orders o ON u.user_id = o.user_id
WHERE o.order_id IS NULL;

--5. Combine all three tables
SELECT u.user_id, u.name, SUM(o.amount) AS total_order_amount, a.login_count
FROM users u 
LEFT JOIN orders o ON u.user_id = o.user_id
LEFT JOIN activity a ON u.user_id = a.user_id
GROUP BY u.user_id, u.name, a.login_count;

--6. Retrieve users who placed order above 400
SELECT DISTINCT u.name FROM users u
JOIN orders o ON u.user_id = o.user_id
WHERE o.amount > 400;

--7. Retrieve top 3 users by total order amount
SELECT u.name, SUM(o.amount) AS total_amount FROM users u
JOIN orders o ON u.user_id = o.user_id
GROUP BY u.user_id, u.name
ORDER BY total_amount DESC 
LIMIT 3;

--8. Retrieve users sorted by total order amount and login count
SELECT u.name, SUM(o.amount) AS total_amount, a.login_count FROM users u
LEFT JOIN orders o ON u.user_id = o.user_id
LEFT JOIN activity a ON u.user_id = a.user_id
GROUP BY u.user_id, u.name, a.login_count
ORDER BY total_amount DESC, a.login_count DESC;

--9. Return the average order amount per user
SELECT u.name, AVG(o.amount) AS average_order_amount
FROM users u
LEFT JOIN orders o ON u.user_id = o.user_id
GROUP BY u.user_id, u.name;

--10. Retrieve top 3 users by total order, login counts
SELECT u.user_id, u.name, SUM(o.amount) AS total_order_amount, a.login_count
FROM users u 
JOIN orders o ON u.user_id = o.user_id
LEFT JOIN activity a ON u.user_id = a.user_id
GROUP BY u.user_id, u.name, a.login_count
ORDER BY total_order_amount DESC
LIMIT 3;
